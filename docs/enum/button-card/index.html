    代码演示

    Studio 风格

    import { useState } from 'react';
    import { ClassicButtonCard } from 'tuya-panel-classic-kit';
    import TuyaRNSvgs from 'tuya-panel-kit/lib/components/iconfont/svg/defaultSvg';
    

    const list = [
      {
        label: '按钮1',
        key: '0',
        disabled: true,
      },
      {
        label: '按钮2',
        key: '1',
      },
      {
        label: '按钮3',
        key: '2',
      },
      {
        label: '按钮4',
        key: '3',
      },
    ];
    

    const list1 = list.map(item => ({
      ...item,
      icon: TuyaRNSvgs.power,
    }));
    

    function Demo() {
      const [activeKeys, setActiveKeys] = useState(['0', '1']);
      const handActiveKeyChange = (key, nextKeys, data) => {
        setActiveKeys(nextKeys);
      };
      return (
        <View>
          <ClassicButtonCard
            title="工作模式"
            showIconBg={false}
            icon={TuyaRNSvgs.power}
            list={list}
            defaultActiveKeys={['1']}
          />
          <Text style={styles.title}>受控制的多选</Text>
          <ClassicButtonCard
            title="工作模式"
            icon={TuyaRNSvgs.power}
            iconSize={14}
            showIconBg
            iconBgColor={{
              deg: 90,
              stops: {
                '0%': 'red',
                '100%': 'yellow',
              },
            }}
            list={list}
            rowCount={4}
            activeKeys={activeKeys}
            activeKeyChange={handActiveKeyChange}
            type="multi"
          />
        </View>
      );
    }

    北欧风格

    import { NordicButtonCard } from 'tuya-panel-nordic-kit';
    import { ClassicIconBackground } from 'tuya-panel-style-icon-background';
    import TuyaRNSvgs from 'tuya-panel-kit/lib/components/iconfont/svg/defaultSvg';
    

    function DemoNordic() {
      const renderButtonItem = data => {
        // console.log(data);
        return (
          <View style={styles.buttonStyle}>
            <ClassicIconBackground showIconBg={false} icon={data.icon} />
            <Text style={{ marginTop: 15 }}>{data.label}</Text>
          </View>
        );
      };
      return (
        <View>
          <NordicButtonCard
            title="工作模式"
            showIconBg={false}
            icon={TuyaRNSvgs.power}
            list={list}
          />
          <Text style={styles.title}>自定义按钮渲染方法</Text>
          <NordicButtonCard
            title="工作模式"
            showIconBg={false}
            icon={TuyaRNSvgs.power}
            list={list1}
            renderButtonItem={renderButtonItem}
            rowCount={4}
            showTitle={false}
            backgroundColor="rgba(255, 255, 255, 0)"
          />
        </View>
      );
    }

    API

    ClassicButtonCard

    属性名描述类型默认值
    list按钮数据RangeItem[](必选)
    activeKeys当前选中的key，传入组件受控ActiveKeys--
    defaultActiveKeys默认选中的 keyActiveKeys--
    activeKeyChange选择key的回调函数(key: string, nextKeys: ActiveKeys, data: RangeItem) => void--
    renderButtonItem自定义的button渲染方法(data: RangeItem) => ReactElement>--
    rowCount每行按钮的个数number--
    type单选模式还是多选模式"radio" | "multi"radio
    disabled是否禁用booleanfalse
    style外层容器样式StyleProp--
    titleContentStyle标题容器样式StyleProp--
    titleStyle标题文字样式StyleProp--
    buttonStyle按钮容器样式StyleProp--
    textStyle按钮文字样式StyleProp--
    padding组件内边距 [上，右，下，左]number[][cx(24), cx(20), cx(24), cx(20)]
    width宽度number--
    backgroundColor组件背景颜色string--
    radius组件圆角number--
    buttonHeight按钮高度number--
    buttonBgColor按钮背景颜色string--
    buttonBgRadius按钮圆角number--
    buttonFontSize按钮文字字体大小number--
    buttonFontColor按钮文字颜色string--
    buttonOffset竖直方向按钮之间的距离number--
    buttonFontWeight按钮文字字重"normal" | "bold" | "100" | "200" | "300" | "400" | "500" | "600" | "700" | "800" | "900"normal
    activeButtonBgColor选中的按钮的背景颜色string--
    activeButtonFontSize选中的按钮的字体大小number--
    activeButtonFontColor选中的按钮字体颜色string--
    activeButtonFontWeight选中的按钮文字字重"normal" | "bold" | "100" | "200" | "300" | "400" | "500" | "600" | "700" | "800" | "900"normal
    title标题string--
    showTitle是否显示标题booleantrue
    titleFontSize标题的文字大小number--
    titleFontColor标题字体颜色string--
    titleFontWeight标题文字字重"normal" | "bold" | "100" | "200" | "300" | "400" | "500" | "600" | "700" | "800" | "900"normal
    icon图标 svg pathstring--
    iconSize图标尺寸numbercx(24)
    iconBgSize组件背景尺寸numbercx(50)
    showIconBg是否显示图标背景booleantrue
    iconColor图标颜色string#fff
    iconBgColor图标背景颜色 只能设置 rgb、rgba 或者 hexBackgroundType#158CFB
    iconBgRadius图标背景圆角numbercx(50)
